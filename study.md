  def cosine_similarity_manual(emb1, emb2):
      dot_product = np.dot(emb1, emb2)
      norm1 = np.linalg.norm(emb1)
      norm2 = np.linalg.norm(emb2)
      return dot_product / (norm1 * norm2)  # 返回0-1之间的相似度
